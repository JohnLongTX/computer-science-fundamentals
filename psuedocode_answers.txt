		INPUT firstNumber
		INPUT secondNumber
		
		FUNCTION greaterNumber(firstNumber, secondNumber)
		  IF firstNumber IS GREATER THAN secondNumber THEN
		    RETURN firstNumber
		  ELSE IF secondNumber IS GREATER THAN firstNumber THEN
		    RETURN secondNumber
		  ENDIF
		END FUNCTION

	3. Create a function that searches for a specific entry in a phonebook.

		INPUT name

		FUNCTION findContact(name, contacts)

		  FOR contact START AT 0 IS LESS THAN contacts.length INCREMENT 1

		    IF name EQUALS contacts[contact].name THEN
		      RETURN contacts[contact].number
		    END IF

		  END FOR

		  RETURN 'Contact not found'

		END FUNCTION
